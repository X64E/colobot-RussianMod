\b;Инструкция \c;buildingenabled\n;
Инструкция \c;buildingenabled();\n; позволяет вам узнать, можно ли построить здание указанной \l;категории\u cbot\category; в течении текущей миссии.

\b;Основное назначение
Инструкция чем-то схожа с \c;\l;canbuild();\u cbot\canbuild;\n;, но она не позовляет проверить, было ли проведено исследование постройки или нет.
\c;
\s; if (buildingenabled(\l;BotFactory\u object\factory;))
\s; {
\s; 	build(\l;BotFactory\u object\factory;);
\s; }
\n;
\b;Для специалистов
Синтаксис:
\s;\c;buildingenabled ( категория );\n;

\t;категория: \c;\l;int\u cbot\int;\n;
\l;Категория\u cbot\category; здания.

\t;Возвращаемое значение: \c;\l;bool\u cbot\bool;\n;
\c;true\n; если здание может быть построено, в противном случае \c;false\n;.

\t;См. также
\c;\l;build\u cbot\build;\n;, \c;\l;canbuild\u cbot\canbuild;\n;.
\l;Programming\u cbot;, \l;types\u cbot\type; and \l;categories\u cbot\category;.
